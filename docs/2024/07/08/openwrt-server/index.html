socks服务

opkg install v2ray-core
vi /etc/cndaqiang/v2ray.json

{
    "inbounds": [
      //可以配置多种协议
        {
            "port": 2090, // 服务器监听端口
            "protocol": "socks",
            "address" : "192.168.192.2"  //地址,可选
	},
        {
            "port": 2091, // 服务器监听端口,和socks使用相同端口,总是很卡
            "protocol": "http",
            "address" : "192.168.192.2" //地址,可选
	}
    ],
    "outbounds": [
      //出口选择freedom就行
        {
            "protocol": "freedom"
        }
    ]
}

开启启动脚本

(sleep 25; /usr/bin/v2ray  run -config  /etc/cndaqiang/v2ray.json &)
